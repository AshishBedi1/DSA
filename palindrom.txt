class Solution {
    public boolean isPalindrome(int x) {
        int number = x;
        int reverse = 0;
        while(x>0){
            int reverseValue = x%10;
        reverse = reverse *10 + reverseValue;

        x = x/10;
        }
        if(reverse == number){
            return true;
        }
        return false;
    }
}
